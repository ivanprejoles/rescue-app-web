import React, { useState, useMemo } from 'react';
import { ChevronDown, ChevronRight } from 'lucide-react';
import { Incident, TypeConfig, ColumnConfig, TableAction } from '../types/incident';

interface IncidentTableProps {
  incidents: Incident[];
  typeConfigs: Record<string, TypeConfig>;
  columns: ColumnConfig[];
  actions?: TableAction[];
  showSummary?: boolean;
  groupBy?: keyof Incident;
  defaultCollapsed?: boolean;
}

const IncidentTable: React.FC<IncidentTableProps> = ({
  incidents,
  typeConfigs,
  columns,
  actions = [],
  showSummary = true,
  groupBy = 'type',
  defaultCollapsed = false
}) => {
  const [collapsedGroups, setCollapsedGroups] = useState<Set<string>>(
    defaultCollapsed ? new Set(Object.keys(typeConfigs)) : new Set()
  );
  const [sortConfig, setSortConfig] = useState<{
    key: string;
    direction: 'asc' | 'desc';
  } | null>(null);

  // Group incidents by the specified field
  const groupedIncidents = useMemo(() => {
    return incidents.reduce((acc, incident) => {
      const groupKey = String(incident[groupBy]);
      if (!acc[groupKey]) {
        acc[groupKey] = [];
      }
      acc[groupKey].push(incident);
      return acc;
    }, {} as Record<string, Incident[]>);
  }, [incidents, groupBy]);

  // Sort incidents within each group
  const sortedGroupedIncidents = useMemo(() => {
    if (!sortConfig) return groupedIncidents;

    const sorted: Record<string, Incident[]> = {};
    Object.entries(groupedIncidents).forEach(([group, incidents]) => {
      sorted[group] = [...incidents].sort((a, b) => {
        const aValue = getNestedValue(a, sortConfig.key);
        const bValue = getNestedValue(b, sortConfig.key);
        
        if (aValue < bValue) return sortConfig.direction === 'asc' ? -1 : 1;
        if (aValue > bValue) return sortConfig.direction === 'asc' ? 1 : -1;
        return 0;
      });
    });
    return sorted;
  }, [groupedIncidents, sortConfig]);

  const getNestedValue = (obj: any, path: string): any => {
    return path.split('.').reduce((current, key) => current?.[key], obj);
  };

  const toggleGroup = (groupKey: string) => {
    const newCollapsed = new Set(collapsedGroups);
    if (newCollapsed.has(groupKey)) {
      newCollapsed.delete(groupKey);
    } else {
      newCollapsed.add(groupKey);
    }
    setCollapsedGroups(newCollapsed);
  };

  const handleSort = (columnKey: string) => {
    const column = columns.find(col => col.key === columnKey);
    if (!column?.sortable) return;

    setSortConfig(current => {
      if (current?.key === columnKey) {
        return current.direction === 'asc' 
          ? { key: columnKey, direction: 'desc' }
          : null;
      }
      return { key: columnKey, direction: 'asc' };
    });
  };

  const renderActionButton = (action: TableAction, incident: Incident, config: TypeConfig) => {
    const baseClasses = "px-3 py-1 text-sm rounded-md transition-all duration-200 font-medium";
    const variantClasses = action.variant === 'primary' 
      ? `${config.color} hover:opacity-90 text-white`
      : "bg-gray-100 hover:bg-gray-200 text-gray-700";
    
    return (
      <button
        key={action.label}
        onClick={() => action.onClick(incident)}
        className={`${baseClasses} ${variantClasses} ${action.className || ''}`}
      >
        {action.label}
      </button>
    );
  };

  return (
    <div className="w-full max-w-7xl mx-auto p-6 bg-gray-50 min-h-screen">
      <div className="mb-8">
        <h1 className="text-3xl font-bold text-gray-900 mb-2">Incident Management</h1>
        <p className="text-gray-600">Track and manage emergency incidents across different categories</p>
      </div>

      <div className="space-y-6">
        {Object.entries(sortedGroupedIncidents).map(([groupKey, groupIncidents]) => {
          const config = typeConfigs[groupKey] || {
            color: 'bg-gray-500',
            lightColor: 'bg-gray-50',
            borderColor: 'border-gray-200',
            textColor: 'text-gray-700',
            icon: () => null,
            label: groupKey
          };
          const Icon = config.icon;
          const isCollapsed = collapsedGroups.has(groupKey);
          
          return (
            <div key={groupKey} className="bg-white rounded-xl shadow-sm border border-gray-200 overflow-hidden">
              {/* Group Header */}
              <div 
                className={`${config.lightColor} px-6 py-4 border-b ${config.borderColor} cursor-pointer hover:opacity-80 transition-opacity`}
                onClick={() => toggleGroup(groupKey)}
              >
                <div className="flex items-center justify-between">
                  <div className="flex items-center space-x-3">
                    <div className="flex items-center space-x-2">
                      {isCollapsed ? (
                        <ChevronRight className="w-4 h-4 text-gray-500" />
                      ) : (
                        <ChevronDown className="w-4 h-4 text-gray-500" />
                      )}
                      <div className={`${config.color} p-2 rounded-lg`}>
                        <Icon className="w-5 h-5 text-white" />
                      </div>
                    </div>
                    <div>
                      <h2 className="text-lg font-semibold text-gray-900">
                        {config.label}
                      </h2>
                      <p className="text-sm text-gray-600">
                        {groupIncidents.length} {groupIncidents.length === 1 ? 'incident' : 'incidents'}
                      </p>
                    </div>
                  </div>
                  <div className={`${config.color} text-white px-3 py-1 rounded-full text-sm font-medium`}>
                    {groupIncidents.length}
                  </div>
                </div>
              </div>

              {/* Table Content */}
              {!isCollapsed && (
                <div className="overflow-x-auto">
                  {/* Table Header */}
                  <div className="bg-gray-50 px-6 py-3 border-b border-gray-200">
                    <div className="flex items-center space-x-4">
                      {columns.map((column) => {
                        const Icon = column.icon;
                        const isSorted = sortConfig?.key === column.key;
                        
                        return (
                          <div
                            key={column.key}
                            className={`flex items-center space-x-1 ${column.width || 'flex-1'} ${
                              column.sortable ? 'cursor-pointer hover:text-gray-900' : ''
                            } text-gray-600 text-sm font-medium`}
                            onClick={() => handleSort(column.key)}
                          >
                            {Icon && <Icon className="w-4 h-4" />}
                            <span>{column.label}</span>
                            {column.sortable && isSorted && (
                              <span className="text-xs">
                                {sortConfig?.direction === 'asc' ? '↑' : '↓'}
                              </span>
                            )}
                          </div>
                        );
                      })}
                      {actions.length > 0 && (
                        <div className="w-32 text-sm font-medium text-gray-600">
                          Actions
                        </div>
                      )}
                    </div>
                  </div>

                  {/* Table Rows */}
                  <div className="divide-y divide-gray-100">
                    {groupIncidents.map((incident) => (
                      <div 
                        key={incident.id} 
                        className="px-6 py-4 hover:bg-gray-50 transition-colors duration-200"
                      >
                        <div className="flex items-center space-x-4">
                          {columns.map((column) => (
                            <div key={column.key} className={column.width || 'flex-1'}>
                              {column.render(incident)}
                            </div>
                          ))}
                          
                          {actions.length > 0 && (
                            <div className="flex items-center space-x-2 w-32">
                              {actions.map((action) => renderActionButton(action, incident, config))}
                            </div>
                          )}
                        </div>
                      </div>
                    ))}
                  </div>
                </div>
              )}
            </div>
          );
        })}
      </div>

      {/* Summary Stats */}
      {showSummary && (
        <div className="mt-8 grid grid-cols-1 md:grid-cols-3 gap-4">
          {Object.entries(groupedIncidents).map(([groupKey, groupIncidents]) => {
            const config = typeConfigs[groupKey] || {
              color: 'bg-gray-500',
              lightColor: 'bg-gray-50',
              borderColor: 'border-gray-200',
              textColor: 'text-gray-700',
              icon: () => null,
              label: groupKey
            };
            const Icon = config.icon;
            
            return (
              <div key={groupKey} className="bg-white rounded-lg p-6 shadow-sm border border-gray-200">
                <div className="flex items-center justify-between">
                  <div>
                    <p className="text-sm text-gray-600">{config.label}</p>
                    <p className="text-2xl font-bold text-gray-900">{groupIncidents.length}</p>
                  </div>
                  <div className={`${config.color} p-3 rounded-lg`}>
                    <Icon className="w-6 h-6 text-white" />
                  </div>
                </div>
              </div>
            );
          })}
        </div>
      )}
    </div>
  );
};

export default IncidentTable;
