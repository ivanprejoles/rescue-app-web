import React, { useState, useMemo } from 'react';
import { ChevronDown, ChevronRight, Plus, MapPin, Users, Building2, Shield } from 'lucide-react';
import { Barangay, EvacuationCenter, BarangayTypeConfig, BarangayColumnConfig, BarangayTableAction } from '../types/barangay';

interface BarangayManagementProps {
  barangays: Barangay[];
  onAddBarangay: () => void;
  onAddEvacuationCenter: (barangayId: string) => void;
  onLocateBarangay: (barangay: Barangay) => void;
  onLocateEvacuationCenter: (center: EvacuationCenter, barangay: Barangay) => void;
  onEditBarangay?: (barangay: Barangay) => void;
  onEditEvacuationCenter?: (center: EvacuationCenter, barangay: Barangay) => void;
  showSummary?: boolean;
}

const BarangayManagement: React.FC<BarangayManagementProps> = ({
  barangays,
  onAddBarangay,
  onAddEvacuationCenter,
  onLocateBarangay,
  onLocateEvacuationCenter,
  onEditBarangay,
  onEditEvacuationCenter,
  showSummary = true
}) => {
  const [collapsedBarangays, setCollapsedBarangays] = useState<Set<string>>(new Set());
  const [sortConfig, setSortConfig] = useState<{
    key: string;
    direction: 'asc' | 'desc';
  } | null>(null);

  const barangayConfig: BarangayTypeConfig = {
    color: 'bg-emerald-500',
    lightColor: 'bg-emerald-50',
    borderColor: 'border-emerald-200',
    textColor: 'text-emerald-700',
    icon: Building2,
    label: 'Barangay'
  };

  const evacuationConfig: BarangayTypeConfig = {
    color: 'bg-blue-500',
    lightColor: 'bg-blue-50',
    borderColor: 'border-blue-200',
    textColor: 'text-blue-700',
    icon: Shield,
    label: 'Evacuation Center'
  };

  const barangayColumns: BarangayColumnConfig[] = [
    {
      key: 'name',
      label: 'Barangay Name',
      icon: Building2,
      render: (barangay: Barangay) => (
        <div className="flex items-center space-x-2">
          <span className="text-sm font-semibold text-gray-900">{barangay.name}</span>
        </div>
      ),
      sortable: true,
      width: 'w-48'
    },
    {
      key: 'captain',
      label: 'Captain',
      icon: Users,
      render: (barangay: Barangay) => (
        <span className="text-sm text-gray-600">
          {barangay.captain || 'Not assigned'}
        </span>
      ),
      sortable: true,
      width: 'w-40'
    },
    {
      key: 'population',
      label: 'Population',
      render: (barangay: Barangay) => (
        <span className="text-sm text-gray-600">
          {barangay.population ? barangay.population.toLocaleString() : 'N/A'}
        </span>
      ),
      sortable: true,
      width: 'w-28'
    },
    {
      key: 'evacuation_centers',
      label: 'Evacuation Centers',
      render: (barangay: Barangay) => (
        <div className="flex items-center space-x-2">
          <span className="text-sm font-medium text-blue-600">
            {barangay.evacuation_centers.length}
          </span>
          <span className="text-xs text-gray-500">centers</span>
        </div>
      ),
      width: 'w-36'
    },
    {
      key: 'location',
      label: 'Location',
      icon: MapPin,
      render: (barangay: Barangay) => (
        <span className="text-sm font-mono text-gray-600">
          {barangay.lat.toFixed(4)}, {barangay.lng.toFixed(4)}
        </span>
      ),
      width: 'w-40'
    }
  ];

  const evacuationColumns: BarangayColumnConfig[] = [
    {
      key: 'name',
      label: 'Center Name',
      icon: Shield,
      render: (center: EvacuationCenter) => (
        <div className="flex items-center space-x-2">
          <span className="text-sm font-medium text-gray-900">{center.name}</span>
        </div>
      ),
      sortable: true,
      width: 'w-48'
    },
    {
      key: 'status',
      label: 'Status',
      render: (center: EvacuationCenter) => {
        const statusColors = {
          active: 'bg-green-100 text-green-800',
          inactive: 'bg-gray-100 text-gray-800',
          maintenance: 'bg-yellow-100 text-yellow-800'
        };
        return (
          <span className={`px-2 py-1 rounded-full text-xs font-medium ${statusColors[center.status]}`}>
            {center.status}
          </span>
        );
      },
      sortable: true,
      width: 'w-28'
    },
    {
      key: 'capacity',
      label: 'Capacity',
      render: (center: EvacuationCenter) => (
        <span className="text-sm text-gray-600">
          {center.capacity ? `${center.capacity} people` : 'N/A'}
        </span>
      ),
      sortable: true,
      width: 'w-32'
    },
    {
      key: 'location',
      label: 'Location',
      icon: MapPin,
      render: (center: EvacuationCenter) => (
        <span className="text-sm font-mono text-gray-600">
          {center.lat.toFixed(4)}, {center.lng.toFixed(4)}
        </span>
      ),
      width: 'w-40'
    }
  ];

  const toggleBarangay = (barangayId: string) => {
    const newCollapsed = new Set(collapsedBarangays);
    if (newCollapsed.has(barangayId)) {
      newCollapsed.delete(barangayId);
    } else {
      newCollapsed.add(barangayId);
    }
    setCollapsedBarangays(newCollapsed);
  };

  const totalEvacuationCenters = useMemo(() => {
    return barangays.reduce((total, barangay) => total + barangay.evacuation_centers.length, 0);
  }, [barangays]);

  const activeEvacuationCenters = useMemo(() => {
    return barangays.reduce((total, barangay) => 
      total + barangay.evacuation_centers.filter(center => center.status === 'active').length, 0
    );
  }, [barangays]);

  return (
    <div className="w-full max-w-7xl mx-auto p-6 bg-gray-50 min-h-screen">
      <div className="mb-8">
        <div className="flex items-center justify-between">
          <div>
            <h1 className="text-3xl font-bold text-gray-900 mb-2">Barangay Management</h1>
            <p className="text-gray-600">Manage barangays and their evacuation centers</p>
          </div>
          <button
            onClick={onAddBarangay}
            className="flex items-center space-x-2 bg-emerald-500 hover:bg-emerald-600 text-white px-4 py-2 rounded-lg transition-colors duration-200 font-medium"
          >
            <Plus className="w-4 h-4" />
            <span>Add Barangay</span>
          </button>
        </div>
      </div>

      <div className="space-y-6">
        {barangays.map((barangay) => {
          const isCollapsed = collapsedBarangays.has(barangay.id);
          
          return (
            <div key={barangay.id} className="bg-white rounded-xl shadow-sm border border-gray-200 overflow-hidden">
              {/* Barangay Header */}
              <div 
                className={`${barangayConfig.lightColor} px-6 py-4 border-b ${barangayConfig.borderColor} cursor-pointer hover:opacity-80 transition-opacity`}
                onClick={() => toggleBarangay(barangay.id)}
              >
                <div className="flex items-center justify-between">
                  <div className="flex items-center space-x-3">
                    <div className="flex items-center space-x-2">
                      {isCollapsed ? (
                        <ChevronRight className="w-4 h-4 text-gray-500" />
                      ) : (
                        <ChevronDown className="w-4 h-4 text-gray-500" />
                      )}
                      <div className={`${barangayConfig.color} p-2 rounded-lg`}>
                        <Building2 className="w-5 h-5 text-white" />
                      </div>
                    </div>
                    <div>
                      <h2 className="text-lg font-semibold text-gray-900">
                        Barangay {barangay.name}
                      </h2>
                      <p className="text-sm text-gray-600">
                        {barangay.evacuation_centers.length} evacuation {barangay.evacuation_centers.length === 1 ? 'center' : 'centers'}
                        {barangay.captain && ` â€¢ Captain: ${barangay.captain}`}
                      </p>
                    </div>
                  </div>
                  <div className="flex items-center space-x-2">
                    <div className={`${barangayConfig.color} text-white px-3 py-1 rounded-full text-sm font-medium`}>
                      {barangay.evacuation_centers.length}
                    </div>
                  </div>
                </div>
              </div>

              {/* Barangay Content */}
              {!isCollapsed && (
                <div>
                  {/* Barangay Details */}
                  <div className="px-6 py-4 bg-gray-50 border-b border-gray-200">
                    <div className="flex items-center space-x-4">
                      {barangayColumns.map((column) => (
                        <div key={column.key} className={column.width || 'flex-1'}>
                          {column.render(barangay)}
                        </div>
                      ))}
                      <div className="flex items-center space-x-2">
                        <button
                          onClick={() => onLocateBarangay(barangay)}
                          className="flex items-center space-x-1 bg-emerald-500 hover:bg-emerald-600 text-white px-3 py-1 rounded-md text-sm transition-colors duration-200"
                        >
                          <MapPin className="w-3 h-3" />
                          <span>Locate</span>
                        </button>
                        {onEditBarangay && (
                          <button
                            onClick={() => onEditBarangay(barangay)}
                            className="bg-gray-100 hover:bg-gray-200 text-gray-700 px-3 py-1 rounded-md text-sm transition-colors duration-200"
                          >
                            Edit
                          </button>
                        )}
                      </div>
                    </div>
                  </div>

                  {/* Evacuation Centers Header */}
                  <div className="px-6 py-3 bg-blue-50 border-b border-blue-200">
                    <div className="flex items-center justify-between">
                      <div className="flex items-center space-x-2">
                        <Shield className="w-4 h-4 text-blue-600" />
                        <span className="text-sm font-medium text-blue-900">Evacuation Centers</span>
                      </div>
                      <button
                        onClick={() => onAddEvacuationCenter(barangay.id)}
                        className="flex items-center space-x-1 bg-blue-500 hover:bg-blue-600 text-white px-3 py-1 rounded-md text-sm transition-colors duration-200"
                      >
                        <Plus className="w-3 h-3" />
                        <span>Add Center</span>
                      </button>
                    </div>
                  </div>

                  {/* Evacuation Centers Table Header */}
                  <div className="px-6 py-3 bg-gray-50 border-b border-gray-200">
                    <div className="flex items-center space-x-4">
                      {evacuationColumns.map((column) => {
                        const Icon = column.icon;
                        return (
                          <div
                            key={column.key}
                            className={`flex items-center space-x-1 ${column.width || 'flex-1'} text-gray-600 text-sm font-medium`}
                          >
                            {Icon && <Icon className="w-4 h-4" />}
                            <span>{column.label}</span>
                          </div>
                        );
                      })}
                      <div className="w-32 text-sm font-medium text-gray-600">
                        Actions
                      </div>
                    </div>
                  </div>

                  {/* Evacuation Centers Rows */}
                  <div className="divide-y divide-gray-100">
                    {barangay.evacuation_centers.length === 0 ? (
                      <div className="px-6 py-8 text-center text-gray-500">
                        <Shield className="w-8 h-8 mx-auto mb-2 text-gray-300" />
                        <p>No evacuation centers added yet</p>
                        <button
                          onClick={() => onAddEvacuationCenter(barangay.id)}
                          className="mt-2 text-blue-600 hover:text-blue-700 text-sm font-medium"
                        >
                          Add the first evacuation center
                        </button>
                      </div>
                    ) : (
                      barangay.evacuation_centers.map((center) => (
                        <div 
                          key={center.id} 
                          className="px-6 py-4 hover:bg-gray-50 transition-colors duration-200"
                        >
                          <div className="flex items-center space-x-4">
                            {evacuationColumns.map((column) => (
                              <div key={column.key} className={column.width || 'flex-1'}>
                                {column.render(center)}
                              </div>
                            ))}
                            
                            <div className="flex items-center space-x-2 w-32">
                              <button
                                onClick={() => onLocateEvacuationCenter(center, barangay)}
                                className="flex items-center space-x-1 bg-blue-500 hover:bg-blue-600 text-white px-2 py-1 rounded text-xs transition-colors duration-200"
                              >
                                <MapPin className="w-3 h-3" />
                                <span>Locate</span>
                              </button>
                              {onEditEvacuationCenter && (
                                <button
                                  onClick={() => onEditEvacuationCenter(center, barangay)}
                                  className="bg-gray-100 hover:bg-gray-200 text-gray-700 px-2 py-1 rounded text-xs transition-colors duration-200"
                                >
                                  Edit
                                </button>
                              )}
                            </div>
                          </div>
                        </div>
                      ))
                    )}
                  </div>
                </div>
              )}
            </div>
          );
        })}

        {barangays.length === 0 && (
          <div className="bg-white rounded-xl shadow-sm border border-gray-200 p-12 text-center">
            <Building2 className="w-12 h-12 mx-auto mb-4 text-gray-300" />
            <h3 className="text-lg font-medium text-gray-900 mb-2">No barangays added yet</h3>
            <p className="text-gray-600 mb-4">Start by adding your first barangay to manage evacuation centers</p>
            <button
              onClick={onAddBarangay}
              className="flex items-center space-x-2 bg-emerald-500 hover:bg-emerald-600 text-white px-4 py-2 rounded-lg transition-colors duration-200 font-medium mx-auto"
            >
              <Plus className="w-4 h-4" />
              <span>Add First Barangay</span>
            </button>
          </div>
        )}
      </div>

      {/* Summary Stats */}
      {showSummary && barangays.length > 0 && (
        <div className="mt-8 grid grid-cols-1 md:grid-cols-3 gap-4">
          <div className="bg-white rounded-lg p-6 shadow-sm border border-gray-200">
            <div className="flex items-center justify-between">
              <div>
                <p className="text-sm text-gray-600">Total Barangays</p>
                <p className="text-2xl font-bold text-gray-900">{barangays.length}</p>
              </div>
              <div className="bg-emerald-500 p-3 rounded-lg">
                <Building2 className="w-6 h-6 text-white" />
              </div>
            </div>
          </div>

          <div className="bg-white rounded-lg p-6 shadow-sm border border-gray-200">
            <div className="flex items-center justify-between">
              <div>
                <p className="text-sm text-gray-600">Evacuation Centers</p>
                <p className="text-2xl font-bold text-gray-900">{totalEvacuationCenters}</p>
              </div>
              <div className="bg-blue-500 p-3 rounded-lg">
                <Shield className="w-6 h-6 text-white" />
              </div>
            </div>
          </div>

          <div className="bg-white rounded-lg p-6 shadow-sm border border-gray-200">
            <div className="flex items-center justify-between">
              <div>
                <p className="text-sm text-gray-600">Active Centers</p>
                <p className="text-2xl font-bold text-gray-900">{activeEvacuationCenters}</p>
              </div>
              <div className="bg-green-500 p-3 rounded-lg">
                <Shield className="w-6 h-6 text-white" />
              </div>
            </div>
          </div>
        </div>
      )}
    </div>
  );
};

export default BarangayManagement;
